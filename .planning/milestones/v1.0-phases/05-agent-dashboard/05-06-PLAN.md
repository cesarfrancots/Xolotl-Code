    1. **AgentMessageList.tsx** — create a virtualized list mirroring MessageList.tsx but sourced from agentStore:
       ```tsx
       import { useRef } from "react";
       import { useVirtualizer } from "@tanstack/react-virtual";
       import { useAgentStore } from "../../stores/agentStore";
       // Reuse the message renderer from the chat module. If MessageList does not export
       // a per-message component, inline a renderer using existing primitives.
       import { MessageItem } from "../chat/Message"; // adjust to actual export name discovered via grep
       // If a streaming-cursor renderer exists separately:
       // import { StreamingMessage } from "../chat/Message";

       export function AgentMessageList({ agentId }: { agentId: string }) {
         const record = useAgentStore(s => s.agents.find(a => a.id === agentId));
         const parentRef = useRef<HTMLDivElement>(null);

         const items = record?.messages ?? [];
         const isStreaming = record?.isStreaming ?? false;
         const streamingContent = record?.streamingContent ?? "";

         // +1 row for the streaming message when active
         const totalCount = items.length + (isStreaming ? 1 : 0);

         const virtualizer = useVirtualizer({
           count: totalCount,
           getScrollElement: () => parentRef.current,
           estimateSize: () => 80,
           measureElement: (el) => el?.getBoundingClientRect().height ?? 80,
           overscan: 5,
         });

         if (!record) {
           return (
             <div className="flex items-center justify-center h-full text-[oklch(0.55_0_0)] text-sm">
               Agent not found.
             </div>
           );
         }

         return (
           <div ref={parentRef} className="h-full overflow-auto px-4 py-2">
             <div style={{ height: virtualizer.getTotalSize(), position: "relative", width: "100%" }}>
               {virtualizer.getVirtualItems().map(vItem => {
                 const isStreamRow = isStreaming && vItem.index === items.length;
                 return (
                   <div
                     key={vItem.key}
                     data-index={vItem.index}
                     ref={virtualizer.measureElement}
                     style={{
                       position: "absolute",
                       top: 0, left: 0, width: "100%",
                       transform: `translateY(${vItem.start}px)`,
                     }}
                   >
                     {isStreamRow ? (
                       <MessageItem
                         message={{
                           id: "__streaming__",
                           role: "assistant",
                           content: streamingContent,
                           toolCalls: [],
                         }}
                         streaming
                       />
                     ) : (
                       (() => {
                         const item = items[vItem.index];
                         // ChatItem can be Message or PermissionItem; AgentMessageList only renders Messages.
                         // Skip non-Message items.
                         if ("role" in item) {
                           return <MessageItem message={item} />;
                         }
                         return null;
                       })()
                     )}
                   </div>
                 );
               })}
             </div>
           </div>
         );
       }
       ```

       export function AgentOutputView({ agentId }: { agentId: string }) {
