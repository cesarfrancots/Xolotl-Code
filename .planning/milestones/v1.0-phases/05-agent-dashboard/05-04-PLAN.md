       ```typescript
       export function useAgentPanelEvents(agentId: string) {
         const deltaBuffer = useRef<string>("");
         const rafId = useRef<number | null>(null);

         useEffect(() => {
           const unlisteners: Array<() => void> = [];
           const channel = `agent-event:${agentId}`;

           const promise = listen<AgentEvent>(channel, (event) => {
             const payload = event.payload;

             if ("TextDelta" in payload) {
               deltaBuffer.current += payload.TextDelta;
               if (rafId.current === null) {
                 rafId.current = requestAnimationFrame(() => {
                   const delta = deltaBuffer.current;
                   deltaBuffer.current = "";
                   rafId.current = null;
                   if (delta) useAgentStore.getState().appendAgentStreamingContent(agentId, delta);
                 });
               }
               return;
             }

             if ("ToolCallStarted" in payload) {
               const { id, tool, input } = payload.ToolCallStarted;
               useAgentStore.getState().startAgentToolCall(agentId, id, tool, input);
               return;
             }

             if ("ToolCallCompleted" in payload) {
               const { id, output } = payload.ToolCallCompleted;
               useAgentStore.getState().completeAgentToolCall(agentId, id, output);
               return;
             }

             if ("TurnCompleted" in payload) {
               // flush any pending rAF before finalizing
               if (rafId.current !== null) {
                 cancelAnimationFrame(rafId.current);
                 rafId.current = null;
                 const delta = deltaBuffer.current;
                 deltaBuffer.current = "";
                 if (delta) useAgentStore.getState().appendAgentStreamingContent(agentId, delta);
               }
               useAgentStore.getState().finalizeAgentStream(agentId, payload.TurnCompleted.usage);
               return;
             }

             if ("StateChanged" in payload) {
               const state = payload.StateChanged;
               useAgentStore.getState().updateAgentState(agentId, state);
               if (state === "Done" || state === "Failed") {
                 void fireDoneNotification(agentId, state);
               }
               return;
             }

             if ("Error" in payload) {
               useAgentStore.getState().appendAgentError(agentId, payload.Error.message);
               return;
             }
           });

           promise.then(unlistenFn => { unlisteners.push(unlistenFn); }).catch(err => {
             console.error(`useAgentPanelEvents listen() failed for ${agentId}:`, err);
           });

           return () => {
             if (rafId.current !== null) {
               cancelAnimationFrame(rafId.current);
               rafId.current = null;
             }
             for (const unlisten of unlisteners) unlisten();
           };
         }, [agentId]);
       }

       // Lazy notification permission cache — request once, reuse.
       let notifPermissionGranted: boolean | null = null;

       async function fireDoneNotification(agentId: string, state: "Done" | "Failed") {
         try {
           if (notifPermissionGranted === null) {
             const granted = await isPermissionGranted();
             notifPermissionGranted = granted || (await requestPermission()) === "granted";
           }
           if (!notifPermissionGranted) return;
           const record = useAgentStore.getState().agents.find(a => a.id === agentId);
           if (!record) return;
           // T-5-03 mitigation: cap title at 60 chars
           const title = record.task.slice(0, 60);
           const cost = record.cumulativeCost.toFixed(4);
           await sendNotification({ title, body: `${state} — $${cost}` });
         } catch (err) {
           console.warn("notification send failed:", err);
         }
       }
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Tauri event payload → React store | Untrusted event payloads written to client state. AgentEvent payloads are produced by the Rust backend, so trust is upstream. |
| sendNotification(title) → OS shell | Task description, untrusted user input, reaches OS notification rendering. WinRT toasts render plain text only. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-5-03 | Spoofing | sendNotification title (user-controlled task string) | mitigate | `record.task.slice(0, 60)` caps the title at 60 chars before passing to sendNotification (Task 2 step 3). WinRT toasts do not render HTML; macOS/Linux similarly treat the title as plain text. |
| T-5-07 | Denial of Service | rAF buffer leak on agent unmount | mitigate | Cleanup callback in useEffect cancels rafId and runs every unlistener (Task 2 step 3). Pattern is exact fork of useAgentEvents.ts which has the same cleanup. |
| T-5-08 | Information Disclosure | Notification fires regardless of window focus (D-14) | accept | Per D-14 user explicitly opted into always-fire. No PII in title/body beyond task + cost. |
</threat_model>
