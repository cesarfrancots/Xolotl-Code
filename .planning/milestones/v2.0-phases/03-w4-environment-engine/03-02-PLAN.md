    ```rust
    /// Pure: deterministically decide whether (and what) the NEXT disaster is, given
    /// (seed, turn) and current env. Returns a CivDisaster whose `remaining_turns` is
    /// the FORECAST LEAD countdown (turns until it fires) — NOT its active duration.
    /// Caller stores it in env.forecast. id/epicenter/radius/intensity all derive from
    /// (seed, turn) so the whole roll is replayable. No uuid, no wall-clock.
    fn roll_forecast(seed: u32, turn: u32, env: &CivEnvironment, world_width: u32) -> Option<CivDisaster> {
        let mut rng = (seed ^ turn.wrapping_mul(0x9E37_79B9) ^ ENV_FORECAST_SALT).max(1);
        // Base chance, modestly higher in harsh seasons / extreme temps (CONTEXT weighting).
        let p = rand_f(&mut rng);
        let chance = match env.season.as_str() { "winter" | "summer" => 0.30, _ => 0.20 };
        if p >= chance { return None; }
        let kinds = disaster_kinds_for(&env.season, env.temperature);
        let kind = kinds[(next_rng(&mut rng) as usize) % kinds.len()];
        let epicenter_x = (1 + (next_rng(&mut rng) % world_width.saturating_sub(2).max(1)))
            .clamp(1, world_width.saturating_sub(2).max(1));
        let radius = (1 + (next_rng(&mut rng) % DISASTER_RADIUS_MAX)).clamp(1, DISASTER_RADIUS_MAX);
        let intensity = round1(rand_range(&mut rng, 0.5, 3.0)).clamp(0.1, 3.0);
        let lead = (DISASTER_FORECAST_LEAD.0
            + (next_rng(&mut rng) % (DISASTER_FORECAST_LEAD.1 - DISASTER_FORECAST_LEAD.0 + 1)))
            .clamp(DISASTER_FORECAST_LEAD.0, DISASTER_FORECAST_LEAD.1);
        Some(CivDisaster {
            id: format!("dis-{turn}-{kind}"),     // seed/turn-derived → replayable, NOT uuid
            kind: kind.to_string(),
            epicenter_x,
            radius,
            intensity,
            remaining_turns: lead,                 // forecast lead countdown (Open Q3 convention)
        })
    }
    ```

    1. Add `apply_disaster_to_tiles`:
    ```rust
    /// Pure: physically reshape `tiles` around the disaster epicenter, in place and
    /// boundedly. flood/quake/landslide convert sub-surface substrate to water/deepwater
    /// (mirrors mining terraform); terrain-neutral kinds (storm/drought/cold_snap/
    /// predator_incursion) leave tiles untouched (their effect is a CivModifier/one-shot).
    /// Invariants: tile count unchanged; x∈[0,width); y∈[WATER_SURFACE_Y, WORLD_HEIGHT);
    /// never converts a tile at/above the seabed surface (keeps the colony floor buildable).
    fn apply_disaster_to_tiles(tiles: &mut [CivTile], dis: &CivDisaster, width: u32) {
        // Terrain-only for the physical-reshape kinds; others are civ-effect/announce.
        if !matches!(dis.kind.as_str(), "flood" | "quake" | "landslide") {
            return;
        }
        let cx = dis.epicenter_x.clamp(1, width.saturating_sub(2).max(1));
        let r = dis.radius.min(DISASTER_RADIUS_MAX);
        let lo = cx.saturating_sub(r);
        let hi = (cx + r).min(width.saturating_sub(1));
        for x in lo..=hi {
            // seabed surface for this column (inline seabed_row_at over the slice)
            let surface = tiles
                .iter()
                .filter(|t| t.x == x && is_substrate(&t.terrain))
                .map(|t| t.y)
                .min()
                .unwrap_or(WORLD_HEIGHT - 2);
            // How deep to reshape this column (bounded by intensity, never the whole column).
            let depth = match dis.kind.as_str() {
                "flood" => 2,           // raise water over the top 1-2 substrate rows
                _ => 1,                 // quake/landslide carve a single sub-surface void
            };
            // Only convert BELOW surface+1 (Pitfall 2: keep the surface solid/buildable).
            let start = (surface + 2).max(WATER_SURFACE_Y);
            for ty in start..(start + depth).min(WORLD_HEIGHT) {
                if let Some(t) = tiles.iter_mut().find(|t| t.x == x && t.y == ty) {
                    if is_substrate(&t.terrain) && ty > surface + 1 && ty >= WATER_SURFACE_Y {
                        t.terrain = if ty >= DEEP_WATER_Y { "deepwater" } else { "water" }.to_string();
                        t.resource = None;
                        t.amount = 0;
                    }
                }
            }
        }
    }
    ```
